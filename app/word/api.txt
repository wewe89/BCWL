API域名:http://app.appwl.cn/api/

注册:appuser/Register
接收参数:
	_n:登录名
	_p:密码
	_m:手机号
返回结果:{"success":true,"message":"","data":1}
success=false(失败):data=错误代码,message=错误提示
		    -1:参数错误
		    -2:登录名已存在
                   -21:手机号已存在
	            -4:未知错误
                    -5:操作异常(追加异常提示)
success=true(成功):data=用户Id

登录:appuser/Login
接收参数:
	_n:登录名
	_p:密码
返回结果:{"success":true,"message":"","data":1}
success=false(失败):data=错误代码,message=错误提示
	 	    -1:参数错误
		    -2:登录失败                   
                    -5:操作异常(追加异常提示)
